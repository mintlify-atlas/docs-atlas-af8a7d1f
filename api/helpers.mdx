## Formatters

### formatCurrency

Formats a number as currency with proper localization.

#### Parameters

<ParamField path="amount" type="number" required={true}>
  The amount to format
</ParamField>

<ParamField path="options" type="CurrencyFormatOptions" required={false}>
  Formatting options
  
  <Expandable title="properties">
    <ResponseField name="currency" type="string" default="'USD'">
      ISO 4217 currency code (e.g., 'USD', 'EUR', 'GBP')
    </ResponseField>
    
    <ResponseField name="locale" type="string" default="'en-US'">
      BCP 47 language tag for localization
    </ResponseField>
    
    <ResponseField name="decimals" type="number" default="2">
      Number of decimal places
    </ResponseField>
  </Expandable>
</ParamField>

#### Return Value

<ResponseField name="formatted" type="string">
  Formatted currency string
</ResponseField>

#### Example

<CodeGroup>
```typescript Usage
import { formatCurrency } from '@/utils/formatters';

formatCurrency(1234.56); // "$1,234.56"
formatCurrency(1234.56, { currency: 'EUR', locale: 'de-DE' }); // "1.234,56 €"
formatCurrency(1234.5, { decimals: 0 }); // "$1,235"
```
</CodeGroup>

### formatDate

Formats a date according to specified format and locale.

#### Parameters

<ParamField path="date" type="Date | string | number" required={true}>
  The date to format (Date object, ISO string, or timestamp)
</ParamField>

<ParamField path="format" type="DateFormat" required={false} default="'medium'">
  Format preset: 'short', 'medium', 'long', 'full', or custom format string
</ParamField>

<ParamField path="options" type="DateFormatOptions" required={false}>
  Additional formatting options
  
  <Expandable title="properties">
    <ResponseField name="locale" type="string" default="'en-US'">
      BCP 47 language tag for localization
    </ResponseField>
    
    <ResponseField name="timeZone" type="string">
      IANA time zone name (e.g., 'America/New_York')
    </ResponseField>
    
    <ResponseField name="includeTime" type="boolean" default="false">
      Include time in the formatted output
    </ResponseField>
  </Expandable>
</ParamField>

#### Return Value

<ResponseField name="formatted" type="string">
  Formatted date string
</ResponseField>

#### Example

<CodeGroup>
```typescript Usage
import { formatDate } from '@/utils/formatters';

const date = new Date('2024-03-15T14:30:00Z');

formatDate(date); // "Mar 15, 2024"
formatDate(date, 'long'); // "March 15, 2024"
formatDate(date, 'short', { includeTime: true }); // "3/15/24, 2:30 PM"
formatDate(date, 'YYYY-MM-DD'); // "2024-03-15"
formatDate(date, 'medium', { locale: 'fr-FR' }); // "15 mars 2024"
```
</CodeGroup>

### validatePassword

Validates password strength against configurable rules.

#### Parameters

<ParamField path="password" type="string" required={true}>
  Password to validate
</ParamField>

<ParamField path="requirements" type="PasswordRequirements" required={false}>
  Password strength requirements
  
  <Expandable title="properties">
    <ResponseField name="minLength" type="number" default="8">
      Minimum password length
    </ResponseField>
    
    <ResponseField name="requireUppercase" type="boolean" default="true">
      Require at least one uppercase letter
    </ResponseField>
    
    <ResponseField name="requireLowercase" type="boolean" default="true">
      Require at least one lowercase letter
    </ResponseField>
    
    <ResponseField name="requireNumbers" type="boolean" default="true">
      Require at least one number
    </ResponseField>
    
    <ResponseField name="requireSpecialChars" type="boolean" default="true">
      Require at least one special character
    </ResponseField>
  </Expandable>
</ParamField>

#### Return Value

<ResponseField name="result" type="ValidationResult">
  Validation result object
  
  <Expandable title="properties">
    <ResponseField name="isValid" type="boolean">
      Whether the password meets all requirements
    </ResponseField>
    
    <ResponseField name="errors" type="string[]">
      Array of validation error messages
    </ResponseField>
    
    <ResponseField name="strength" type="'weak' | 'medium' | 'strong'">
      Password strength rating
    </ResponseField>
  </Expandable>
</ResponseField>

#### Example

<CodeGroup>
```typescript Usage
import { validatePassword } from '@/utils/validators';

validatePassword('Pass123!');
// { isValid: true, errors: [], strength: 'strong' }

validatePassword('weak');
// { 
//   isValid: false, 
//   errors: ['Must be at least 8 characters', 'Must contain a number'],
//   strength: 'weak'
// }

validatePassword('simple123', { requireSpecialChars: false });
// { isValid: true, errors: [], strength: 'medium' }
```
</CodeGroup>

## Data Manipulation

### debounce

Creates a debounced function that delays execution.

#### Parameters

<ParamField path="func" type="Function" required={true}>
  The function to debounce
</ParamField>

<ParamField path="delay" type="number" required={true}>
  Delay in milliseconds
</ParamField>

#### Return Value

<ResponseField name="debouncedFunc" type="Function & { cancel: () => void }">
  Debounced function with a cancel method
</ResponseField>

#### Example

<CodeGroup>
```typescript Usage
import { debounce } from '@/utils/helpers';

const handleSearch = debounce((query: string) => {
  console.log('Searching for:', query);
}, 500);

// Only executes once after 500ms of no calls
handleSearch('a');
handleSearch('ab');
handleSearch('abc'); // Only this will execute

// Cancel pending execution
handleSearch.cancel();
```
</CodeGroup>

### groupBy

Groups array items by a specified key or function.

#### Parameters

<ParamField path="array" type="T[]" required={true}>
  Array to group
</ParamField>

<ParamField path="keyOrFn" type="keyof T | ((item: T) => string)" required={true}>
  Property key or function to group by
</ParamField>

#### Return Value

<ResponseField name="grouped" type="Record<string, T[]>">
  Object with grouped items
</ResponseField>

#### Example

<CodeGroup>
```typescript Usage
import { groupBy } from '@/utils/helpers';

const users = [
  { name: 'Alice', role: 'admin' },
  { name: 'Bob', role: 'user' },
  { name: 'Charlie', role: 'admin' }
];

const byRole = groupBy(users, 'role');
// {
//   admin: [{ name: 'Alice', role: 'admin' }, { name: 'Charlie', role: 'admin' }],
//   user: [{ name: 'Bob', role: 'user' }]
// }

const byFirstLetter = groupBy(users, user => user.name[0]);
// {
//   A: [{ name: 'Alice', role: 'admin' }],
//   B: [{ name: 'Bob', role: 'user' }],
//   C: [{ name: 'Charlie', role: 'admin' }]
// }
```
</CodeGroup>

### sortBy

Sorts array by specified key or comparator function.

#### Parameters

<ParamField path="array" type="T[]" required={true}>
  Array to sort
</ParamField>

<ParamField path="keyOrFn" type="keyof T | ((item: T) => any)" required={true}>
  Property key or comparator function
</ParamField>

<ParamField path="order" type="'asc' | 'desc'" required={false} default="'asc'">
  Sort order
</ParamField>

#### Return Value

<ResponseField name="sorted" type="T[]">
  New sorted array (original unchanged)
</ResponseField>

#### Example

<CodeGroup>
```typescript Usage
import { sortBy } from '@/utils/helpers';

const products = [
  { name: 'Phone', price: 699 },
  { name: 'Laptop', price: 1299 },
  { name: 'Tablet', price: 499 }
];

const byPrice = sortBy(products, 'price');
// [{ name: 'Tablet', price: 499 }, { name: 'Phone', price: 699 }, ...]

const byPriceDesc = sortBy(products, 'price', 'desc');
// [{ name: 'Laptop', price: 1299 }, { name: 'Phone', price: 699 }, ...]

const byName = sortBy(products, p => p.name.toLowerCase());
```
</CodeGroup>
