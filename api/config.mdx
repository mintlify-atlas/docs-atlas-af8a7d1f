## App Configuration

### createAppConfig

Creates and initializes the application configuration.

#### Parameters

<ParamField path="options" type="AppConfigOptions" required={false}>
  Configuration options to override defaults
  
  <Expandable title="properties">
    <ResponseField name="apiUrl" type="string">
      Base URL for API requests
    </ResponseField>
    
    <ResponseField name="environment" type="'development' | 'staging' | 'production'">
      Current environment
    </ResponseField>
    
    <ResponseField name="features" type="FeatureFlags">
      Feature flag configuration
    </ResponseField>
    
    <ResponseField name="auth" type="AuthConfig">
      Authentication configuration
    </ResponseField>
    
    <ResponseField name="theme" type="ThemeConfig">
      Theme and UI configuration
    </ResponseField>
    
    <ResponseField name="analytics" type="AnalyticsConfig">
      Analytics and tracking configuration
    </ResponseField>
  </Expandable>
</ParamField>

#### Return Value

<ResponseField name="config" type="AppConfig">
  Initialized application configuration object
</ResponseField>

#### Example

<CodeGroup>
```typescript Basic Usage
import { createAppConfig } from '@/config';

const config = createAppConfig({
  apiUrl: process.env.NEXT_PUBLIC_API_URL,
  environment: process.env.NODE_ENV as 'development' | 'production',
  features: {
    enableBetaFeatures: false,
    enableAnalytics: true
  }
});

export default config;
```

```typescript Full Configuration
import { createAppConfig } from '@/config';

const config = createAppConfig({
  apiUrl: 'https://api.example.com',
  environment: 'production',
  
  features: {
    enableBetaFeatures: false,
    enableAnalytics: true,
    enableNotifications: true,
    enableDarkMode: true
  },
  
  auth: {
    tokenKey: 'auth_token',
    refreshTokenKey: 'refresh_token',
    sessionTimeout: 3600000,
    autoRefresh: true
  },
  
  theme: {
    defaultMode: 'light',
    primaryColor: '#3B82F6',
    fontFamily: 'Inter, sans-serif'
  },
  
  analytics: {
    enabled: true,
    trackingId: 'GA-XXXXXXXXX',
    debug: false
  }
});

export default config;
```
</CodeGroup>

<ResponseField name="FeatureFlags" type="object">

    <ResponseField name="enableBetaFeatures" type="boolean" default="false">

    </ResponseField>
    
    <ResponseField name="enableAnalytics" type="boolean" default="true">
      Enable analytics tracking
    </ResponseField>
    
    <ResponseField name="enableNotifications" type="boolean" default="true">
      Enable push notifications
    </ResponseField>
    
    <ResponseField name="enableDarkMode" type="boolean" default="true">
      Enable dark mode support
    </ResponseField>
    
    <ResponseField name="enableSocialAuth" type="boolean" default="true">
      Enable social authentication providers
    </ResponseField>
    
    <ResponseField name="enableFileUpload" type="boolean" default="true">
      Enable file upload functionality
    </ResponseField>
    
    <ResponseField name="enableRealtime" type="boolean" default="false">
      Enable real-time WebSocket features
    </ResponseField>
    
    <ResponseField name="enableExport" type="boolean" default="true">
      Enable data export functionality
    </ResponseField>
  </Expandable>
</ResponseField>

#### Example

<CodeGroup>

function DashboardPage() {

## Authentication Config

### AuthConfig

Authentication configuration settings.

<ResponseField name="AuthConfig" type="object">
  <Expandable title="properties">
    <ResponseField name="tokenKey" type="string" default="'auth_token'">
      LocalStorage key for storing auth token
    </ResponseField>
    
    <ResponseField name="refreshTokenKey" type="string" default="'refresh_token'">
      LocalStorage key for storing refresh token
    </ResponseField>
    
    <ResponseField name="sessionTimeout" type="number" default="3600000">
      Session timeout in milliseconds (default: 1 hour)
    </ResponseField>
    
    <ResponseField name="autoRefresh" type="boolean" default="true">
      Automatically refresh tokens before expiry
    </ResponseField>
    
    <ResponseField name="refreshThreshold" type="number" default="300000">
      Time before expiry to refresh token in ms (default: 5 minutes)
    </ResponseField>
    
    <ResponseField name="loginRedirect" type="string" default="'/dashboard'">
      URL to redirect to after successful login
    </ResponseField>
    
    <ResponseField name="logoutRedirect" type="string" default="'/login'">
      URL to redirect to after logout
    </ResponseField>
  </Expandable>
</ResponseField>

#### Example

<CodeGroup>
```typescript Custom Auth Config
import { createAppConfig } from '@/config';

const config = createAppConfig({
  auth: {
    tokenKey: 'app_token',
    refreshTokenKey: 'app_refresh',
    sessionTimeout: 7200000, // 2 hours
    autoRefresh: true,
    refreshThreshold: 600000, // 10 minutes
    loginRedirect: '/home',
    logoutRedirect: '/'
  }
});
```
</CodeGroup>

## API Configuration

### ApiConfig

API client configuration settings.

<ResponseField name="ApiConfig" type="object">
  <Expandable title="properties">
    <ResponseField name="baseURL" type="string">
      Base URL for all API requests
    </ResponseField>
    
    <ResponseField name="timeout" type="number" default="30000">
      Request timeout in milliseconds
    </ResponseField>
    
    <ResponseField name="retries" type="number" default="3">
      Number of retry attempts for failed requests
    </ResponseField>
    
    <ResponseField name="retryDelay" type="number" default="1000">
      Delay between retries in milliseconds
    </ResponseField>
    
    <ResponseField name="headers" type="Record<string, string>">
      Default headers for all requests
    </ResponseField>
    
    <ResponseField name="withCredentials" type="boolean" default="false">
      Send cookies with cross-origin requests
    </ResponseField>
  </Expandable>
</ResponseField>

#### Example

<CodeGroup>
```typescript API Client Config
import { createAppConfig } from '@/config';

const config = createAppConfig({
  api: {
    baseURL: process.env.NEXT_PUBLIC_API_URL,
    timeout: 30000,
    retries: 3,
    retryDelay: 1000,
    headers: {
      'Content-Type': 'application/json',
      'X-App-Version': '1.0.0'
    },
    withCredentials: true
  }
});
```
</CodeGroup>

## Updating Config at Runtime

### updateConfig

Updates configuration values at runtime.

#### Parameters

<ParamField path="updates" type="Partial<AppConfig>" required={true}>
  Configuration values to update
</ParamField>

#### Example

<CodeGroup>
```typescript Dynamic Config Update
import { updateConfig } from '@/config';

function SettingsPage() {
  const toggleDarkMode = () => {
    updateConfig({
      theme: {
        defaultMode: 'dark'
      }
    });
  };
  
  const enableBetaFeatures = () => {
    updateConfig({
      features: {
        enableBetaFeatures: true
      }
    });
  };
  
  return (
    <div>
      <button onClick={toggleDarkMode}>Enable Dark Mode</button>
      <button onClick={enableBetaFeatures}>Enable Beta Features</button>
    </div>
  );
}
```
</CodeGroup>
